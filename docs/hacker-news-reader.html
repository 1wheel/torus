A date formatter that does relative dates in English for the last 2 days.


29const formatDate = unix => {


30    if (!unix) return 'unknown';


31


32    const date = new Date(unix * 1000);


33    const delta = (NOW - date) / 1000;


34    if (delta < 60) {


35        return '&#60;1 min';


36    } else if (delta < 3600) {


37        return `${~~(delta / 60)} min`;


38    } else if (delta < 86400) {


39        return `${~~(delta / 3600)} hr`;


40    } else if (delta < 86400 * 2) {


41        return 'yesterday';


42    } else if (delta < 86400 * 3) {


43        return '2 days ago';


44    } else {


45        return date.toLocaleDateString() + ' ' + formatTime(date);


46    }


47}
