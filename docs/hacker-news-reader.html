11const formatTime = date => {


12    const pad = num => num.toString().padStart(2, '0');

16const formatDate = unix => {


17    if (!unix) return 'unknown';


18


19    const date = new Date(unix * 1000);


20    const delta = (NOW - date) / 1000;


21    if (delta < 60) {


22        return '&#60;1 min';


23    } else if (delta < 3600) {


24        return `${~~(delta / 60)} min`;


25    } else if (delta < 86400) {


26        return `${~~(delta / 3600)} hr`;


27    } else if (delta < 86400 * 2) {


28        return 'yesterday';


29    } else if (delta < 86400 * 3) {


30        return '2 days ago';


31    } else {


32        return date.toLocaleDateString() + ' ' + formatTime(date);


33    }


34}

49class Item extends Record {


50


51    /* Items have the following attrs we care about:


52     *  id: number


53     *  deleted: boolean


54     *  type: 'job', 'story', 'comment', 'poll/pollopt' (which we ignore)


55     *  by: username in string


56     *  time: unix


57     *  text: text content


58     *  dead: boolean


59     *  parent: id of parent story or comment


60     *  kids: kids in display order ranked


61     *  url: string


62     *  score: number of votes, or #votes for pollopt


63     *  title: string


64     *  descendants: total comment count


65     */


66    fetch() {


67        if (!this.loaded) {


68            hnFetch(`/item/${this.id}`).then(data => {


69                const { id, ...attrs } = data;


70                this.update(attrs);


71            });


72        }


73    }


74


75    get loaded() {


76        return this.get('type');


77    }


78


79}

83class StoryStore extends StoreOf(Story) {


84


85    constructor(slug, limit = 25) {


86        super();


87        this.slug = slug;


88        this.limit = limit;


89        this.start = 0;


90    }


91


92    fetch() {


93        hnFetch('/' + this.slug).then(stories => {


94            const storyRecords = stories.slice(


95                this.start * this.limit,


96                (this.start + 1) * this.limit


97            ).map(id => new Story(id));


98            this.reset(storyRecords);


99            for (const story of storyRecords) {


100                story.fetch();


101            }


102        });


103    }


104


105    nextPage() {


106        this.start ++;


107        this.fetch();


108    }


109


110    previousPage() {


111        this.start = Math.max(0, this.start - 1);


112        this.fetch();


113    }


114


115}
